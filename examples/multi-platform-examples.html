import React, { useEffect, useRef } from 'react'; import { EnhancedAgentWidget } from 'ttp-agent-sdk'; function VoiceWidget() { const widgetRef = useRef(null); useEffect(() => { if (widgetRef.current) return; widgetRef.current = new EnhancedAgentWidget({ agentId: 'your_agent_id', getSessionUrl: async ({ agentId, variables }) => { const response = await fetch('/api/get-voice-session', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ agentId, variables }) }); const data = await response.json(); return data.signedUrl; }, icon: { type: 'custom', customImage: '/logo.png', size: 'large' }, button: { primaryColor: '#4f46e5', hoverColor: '#4338ca' }, position: { vertical: 'bottom', horizontal: 'right' }, variables: { userId: 'user123', page: 'homepage' } }); return () => { if (widgetRef.current) { widgetRef.current.destroy(); widgetRef.current = null; } }; }, []); return null; // Widget renders itself } export default VoiceWidget;

<!DOCTYPE html> <html> <head> <title>Voice App</title> </head> <body> <div id="app"> <h1>Voice Chat</h1> <button id="connectBtn">Connect</button> <button id="recordBtn" disabled>Start Recording</button> <div id="messages"></div> </div> <script src="https://cdn.talktopc.com/ttp-agent-sdk/agent-widget.js"></script> <script> class VoiceApp { constructor() { this.isConnected = false; this.isRecording = false; this.messages = []; this.voiceSDK = null; this.init(); } init() { this.voiceSDK = new TTPAgentSDK.VoiceSDK({ websocketUrl: 'wss://speech.talktopc.com/ws/conv', agentId: 'your_agent_id', appId: 'your_app_id' }); this.setupEventHandlers(); this.setupUI(); } setupEventHandlers() { this.voiceSDK.on('connected', () => { this.isConnected = true; this.updateUI(); this.addMessage('system', 'Connected to voice agent'); }); this.voiceSDK.on('disconnected', () => { this.isConnected = false; this.isRecording = false; this.updateUI(); this.addMessage('system', 'Disconnected'); }); this.voiceSDK.on('recordingStarted', () => { this.isRecording = true; this.updateUI(); this.addMessage('user', 'üé§ Recording...'); }); this.voiceSDK.on('recordingStopped', () => { this.isRecording = false; this.updateUI(); }); this.voiceSDK.on('message', (message) => { if (message.type === 'agent_response') { this.addMessage('agent', message.agent_response); } }); } setupUI() { document.getElementById('connectBtn').onclick = () => this.connect(); document.getElementById('recordBtn').onclick = () => this.toggleRecording(); } async connect() { try { await this.voiceSDK.connect(); } catch (error) { this.addMessage('error', `Connection failed: ${error.message}`); } } async toggleRecording() { try { if (this.isRecording) { await this.voiceSDK.stopRecording(); } else { await this.voiceSDK.startRecording(); } } catch (error) { this.addMessage('error', `Recording failed: ${error.message}`); } } addMessage(type, text) { this.messages.push({ type, text, timestamp: new Date() }); this.renderMessages(); } renderMessages() { const container = document.getElementById('messages'); container.innerHTML = this.messages.map(msg => ` <div class="message ${msg.type}"> <strong>${msg.type}:</strong> ${msg.text} <small>${msg.timestamp.toLocaleTimeString()}</small> </div> `).join(''); } updateUI() { const connectBtn = document.getElementById('connectBtn'); const recordBtn = document.getElementById('recordBtn'); connectBtn.disabled = this.isConnected; connectBtn.textContent = this.isConnected ? 'Connected' : 'Connect'; recordBtn.disabled = !this.isConnected; recordBtn.textContent = this.isRecording ? 'Stop Recording' : 'Start Recording'; } } // Initialize app new VoiceApp(); </script> </body> </html>
VoiceChat.jsx
import React, { useState, useEffect, useRef } from 'react'; import { VoiceSDK } from 'ttp-agent-sdk'; function VoiceChat() { const [isConnected, setIsConnected] = useState(false); const [isRecording, setIsRecording] = useState(false); const [messages, setMessages] = useState([]); const voiceSDKRef = useRef(null); useEffect(() => { const voiceSDK = new VoiceSDK({ websocketUrl: 'wss://speech.talktopc.com/ws/conv', agentId: 'your_agent_id', appId: 'your_app_id' }); voiceSDK.on('connected', () => setIsConnected(true)); voiceSDK.on('disconnected', () => setIsConnected(false)); voiceSDK.on('recordingStarted', () => setIsRecording(true)); voiceSDK.on('recordingStopped', () => setIsRecording(false)); voiceSDK.on('message', (message) => { if (message.type === 'agent_response') { setMessages(prev => [...prev, { type: 'agent', text: message.agent_response }]); } }); voiceSDKRef.current = voiceSDK; return () => voiceSDK.destroy(); }, []); const handleConnect = async () => { await voiceSDKRef.current.connect(); }; const handleToggleRecording = async () => { if (isRecording) { await voiceSDKRef.current.stopRecording(); } else { await voiceSDKRef.current.startRecording(); } }; return ( <div> <button onClick={handleConnect} disabled={isConnected}> {isConnected ? 'Connected' : 'Connect'} </button> <button onClick={handleToggleRecording} disabled={!isConnected}> {isRecording ? 'Stop Recording' : 'Start Recording'} </button> <div> {messages.map((msg, i) => ( <div key={i}><strong>{msg.type}:</strong> {msg.text}</div> ))} </div> </div> ); } export default VoiceChat;
